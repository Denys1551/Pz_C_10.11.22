#include <iostream>
#include <math.h>
#include <cmath>
using namespace std;

int main(void) {
    
    const int size = 12;
    int m[size] = {10,8,9,9,11,5,8,7,7,6,12,10};
    
    for (int i = 0; i < size; i++)
    {
        cout << m[i] << " ";
    }
    cout << endl;
    
    int input;
    cout << "Enter num: ";
    cin >> input;

    for (int i = 0; i < size; i++)
    {
        if (m[i] > input)
        {
            cout << m[i] << " ";
        }
    } 
}
